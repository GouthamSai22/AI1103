\section{Problem(prob. misc. 6.11)}
Given that the events A and B are such that P(A) = $\frac{1}{2}$, P(A + B) = $\frac{3}{5}$ and P(B) = p. Find p if they are 
\begin{enumerate}[label={\roman*)}]
    \item mutually exclusive
    \item independent 
\end{enumerate}
\section{Solution}
i)
Since the events are mutually exclusive, by definition
\begin{align}
    P(AB) = 0\\
    \implies P(A + B) = P(A) + P(B)\label{Zehahahaha}
\end{align}
On substituting the values of P(A), P(B) and P(A$\cup$B) in \eqref{Zehahahaha}, we get
\begin{align}
    \cfrac{3}{5} = \cfrac{1}{2} + p\\
    \implies p = \cfrac{1}{10}
\end{align}
ii)
Since the events are independent
\begin{align}
    P(AB) = P(A)P(B)
\end{align}
We know
\begin{align}
    P(A + B) = P(A) + P(B) - P(AB)\\
    \implies P(A + B) = P(A) + P(B) - P(A)P(B)\label{Gababababa}
\end{align}
On substituting the values of P(A), P(B) and P(A$\cup$B) in \eqref{Gababababa}, we get
\begin{align}
    \cfrac{3}{5} = \cfrac{1}{2} + p - \cfrac{1}{2} p\\
    \implies p = \cfrac{1}{5}
\end{align}
